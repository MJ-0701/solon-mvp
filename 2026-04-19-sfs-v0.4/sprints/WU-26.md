## §1. `/sfs decision` 구현 spec (mini-ADR 신설)

### 1.1 Output 형식 확정

```
decision created: decisions/<id>-<kebab>.md
```

(stdout 1줄. 에디터 launch 안 함, WU-25 §1.1 plan 패턴 정합.)

### 1.2 Bash 골격 (sfs-decision.sh draft)

```sh
#!/usr/bin/env bash
# sfs-decision.sh — /sfs decision dispatch (WU-26 §1)
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
. "$SCRIPT_DIR/sfs-common.sh"

TITLE=""; OVERRIDE_ID=""
while [ $# -gt 0 ]; do
    case "$1" in
        --id) OVERRIDE_ID="$2"; shift 2;;
        --id=*) OVERRIDE_ID="${1#--id=}"; shift;;
        -*) die 7 "unknown arg: $1";;
        *) [ -z "$TITLE" ] && TITLE="$1" || die 7 "extra arg: $1"; shift;;
    esac
done
[ -n "$TITLE" ] || die 7 "usage: /sfs decision <title> [--id <id>]"

validate_sfs_local

# (a) decision id 자동 부여 또는 override
if [ -n "$OVERRIDE_ID" ]; then
    DECISION_ID="$OVERRIDE_ID"
    [ ! -f ".sfs-local/decisions/$DECISION_ID-"*.md ] || die 1 "decision id $DECISION_ID already exists"
else
    DECISION_ID="$(next_decision_id)"   # 가장 큰 기존 id + 1, 시작 0001
fi

# (b) kebab-case slug from title
KEBAB="$(echo "$TITLE" | tr '[:upper:]' '[:lower:]' | tr -c 'a-z0-9' '-' | sed 's/--*/-/g; s/^-//; s/-$//')"
[ -n "$KEBAB" ] || die 7 "title produces empty kebab-case slug"

DECISIONS_DIR=".sfs-local/decisions"
mkdir -p "$DECISIONS_DIR"
TEMPLATE=".sfs-local/decisions-template/ADR-TEMPLATE.md"
[ -f "$TEMPLATE" ] || die 4 "template missing: $TEMPLATE"

DECISION_PATH="$DECISIONS_DIR/$DECISION_ID-$KEBAB.md"
NOW="$(date -u +%Y-%m-%dT%H:%M:%S+00:00)"
cp "$TEMPLATE" "$DECISION_PATH"
sed -i.bak "s/{{DECISION_ID}}/$DECISION_ID/g; s/{{TITLE}}/$TITLE/g; s/{{NOW}}/$NOW/g" "$DECISION_PATH" && rm -f "$DECISION_PATH.bak"

# (c) events.jsonl append
append_event "{\"type\":\"decision_created\",\"decision_id\":\"$DECISION_ID\",\"path\":\"$DECISION_PATH\",\"title\":\"$TITLE\",\"ts\":\"$NOW\"}"

# (d) stdout
echo "decision created: $DECISION_PATH"
```

| exit | 조건 | stderr |
|:-:|---|---|
| 0 | 정상 | (없음) |
| 1 | `--id` 충돌 | `decision id <id> already exists` |
| 4 | template 부재 | `template missing: <path>` |
| 7 | `<title>` 미지정 또는 unknown CLI flag | usage 또는 `unknown arg: <flag>` |

## §2. `/sfs retro --close` 구현 spec (sprint close + auto commit)

### 2.1 Output 형식 확정

```
retro.md ready: <path>
sprint closed: <sprint-id>     # --close 인 경우만 추가 출력
```

(`--close` 미지정 시 retro.md 진입만, stdout 1줄. `--close` 지정 시 sprint close + auto commit + stdout 2줄.)

### 2.2 Bash 골격 (sfs-retro.sh draft)

```sh
#!/usr/bin/env bash
# sfs-retro.sh — /sfs retro [--close] dispatch (WU-26 §2)
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
. "$SCRIPT_DIR/sfs-common.sh"

CLOSE=0
while [ $# -gt 0 ]; do
    case "$1" in
        --close) CLOSE=1; shift;;
        *) die 7 "unknown arg: $1";;
    esac
done

validate_sfs_local
SPRINT_ID="$(read_current_sprint)"
[ -n "$SPRINT_ID" ] || die 1 "no active sprint, run /sfs start first"

SPRINT_DIR=".sfs-local/sprints/$SPRINT_ID"
RETRO_PATH="$SPRINT_DIR/retro.md"
REVIEW_PATH="$SPRINT_DIR/review.md"

# (a) retro.md 미존재 시 template 적용
if [ ! -f "$RETRO_PATH" ]; then
    TEMPLATE=".sfs-local/sprint-templates/retro.md"
    [ -f "$TEMPLATE" ] || die 4 "template missing: $TEMPLATE"
    cp "$TEMPLATE" "$RETRO_PATH"
    sed -i.bak "s/{{SPRINT_ID}}/$SPRINT_ID/g" "$RETRO_PATH" && rm -f "$RETRO_PATH.bak"
fi

# (b) frontmatter 갱신
NOW="$(date -u +%Y-%m-%dT%H:%M:%S+00:00)"
update_frontmatter "$RETRO_PATH" "phase" "retro"
update_frontmatter "$RETRO_PATH" "last_touched_at" "$NOW"
append_event "{\"type\":\"retro_open\",\"sprint_id\":\"$SPRINT_ID\",\"path\":\"$RETRO_PATH\",\"ts\":\"$NOW\"}"

echo "retro.md ready: $RETRO_PATH"

# (c) --close 분기
if [ "$CLOSE" -eq 1 ]; then
    [ -f "$REVIEW_PATH" ] || die 8 "review.md required before close (run /sfs review first)"

    # sprint frontmatter status=closed + closed_at (sprint dir 의 first .md 파일 = plan.md 또는 별도 sprint.md)
    sprint_close "$SPRINT_DIR" "$NOW"   # sfs-common.sh 신설 함수

    # current-sprint 클리어
    rm -f .sfs-local/current-sprint
    append_event "{\"type\":\"sprint_close\",\"sprint_id\":\"$SPRINT_ID\",\"ts\":\"$NOW\"}"

    # auto commit (사용자 명시 호출 = §1.5' 정합, push 안 함)
    auto_commit_close "$SPRINT_ID"   # sfs-common.sh 신설 함수

    echo "sprint closed: $SPRINT_ID"
fi
```

## §3. 공통 helper `sfs-common.sh` 보강 (WU-25 §3 위에 추가)

```sh
# next_decision_id — .sfs-local/decisions/ 의 가장 큰 4-자리 id + 1
next_decision_id() {
    local dir=".sfs-local/decisions"
    [ -d "$dir" ] || { echo "0001"; return 0; }
    local max=$(ls -1 "$dir" 2>/dev/null | grep -oE '^[0-9]{4}' | sort -n | tail -1)
    if [ -z "$max" ]; then echo "0001"; else printf '%04d\n' $((10#$max + 1)); fi
}

# sprint_close <sprint-dir> <iso-ts> — sprint frontmatter status=closed + closed_at
sprint_close() {
    local sdir="$1" ts="$2"
    # plan.md frontmatter 갱신 (또는 별도 sprint.md 가 있으면 그것)
    local target="$sdir/plan.md"
    [ -f "$target" ] || target="$sdir/sprint.md"
    [ -f "$target" ] || return 0   # frontmatter target 부재 시 silent
    update_frontmatter "$target" "status" "closed"
    update_frontmatter "$target" "closed_at" "$ts"
}

# auto_commit_close <sprint-id> — sprint close 후 git add + commit (push 안 함)
# WARNING: AI 자율 호출 금지 (§1.5' 정합), 사용자가 sfs retro --close 명시 호출 시에만 동작.
auto_commit_close() {
    local sid="$1"
    git add ".sfs-local/sprints/$sid/" ".sfs-local/events.jsonl" 2>/dev/null || true
    [ ! -f ".sfs-local/current-sprint" ] && git add ".sfs-local/current-sprint" 2>/dev/null || true
    git commit -m "chore(sfs): close sprint $sid" 2>/dev/null || true
    # push 는 §1.5 정합 사용자 manual
}
```
